Backend Spec: Recipe Ingredients with Spoonacular Categorization

  Overview

  Enhance recipe functionality with parsed ingredients and automatic categorization using Spoonacular API. Implement global caching to minimize API usage and ensure fast lookups.

  ---
  1. Environment Configuration

  Add to .env:
  SPOONACULAR_API_KEY=your_api_key_here

  Get API key from: https://spoonacular.com/food-api/console#Dashboard

  2. Updated Data Models

  Recipe (Enhanced)

    ingredient: string;           // "chicken breast" (normalized, lowercase, PRIMARY KEY)
    category: string;             // "Meat"

  3. Database Schema (SQL)

  -- Enhanced recipes table
  CREATE TABLE recipes (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    url TEXT NOT NULL,
    title TEXT NOT NULL,
    image_url TEXT,
    description TEXT,
    ingredients JSONB NOT NULL DEFAULT '[]',          -- Array of strings
    parsed_ingredients JSONB NOT NULL DEFAULT '[]',   -- Array of ParsedIngredient objects
    instructions JSONB NOT NULL DEFAULT '[]',
    prep_time TEXT,
    cook_time TEXT,
    total_time TEXT,
    servings TEXT,
    cuisine TEXT,
    category TEXT,
    notes TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
  );

  -- Global ingredient categories cache
  CREATE TABLE ingredient_categories (
    ingredient VARCHAR(255) PRIMARY KEY,  -- normalized lowercase
    category VARCHAR(100) NOT NULL,
    aisle VARCHAR(100),
    spoonacular_id INTEGER,
    image_url TEXT,
    source VARCHAR(20) NOT NULL DEFAULT 'spoonacular',
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
  );

  CREATE INDEX idx_ingredient_category ON ingredient_categories(category);
  CREATE INDEX idx_ingredient_source ON ingredient_categories(source);

  5. Ingredient Parsing & Categorization Logic

  Step-by-Step Process

  When importing a recipe:

  1. Parse each ingredient string using ingredient-parser-nlp or similar:
  const parser = require('ingredient-parser-nlp');

  // Input: "2 cups all-purpose flour, sifted"
  const parsed = parser.parse("2 cups all-purpose flour, sifted");
  // {
  //   quantity: 2,
  //   unit: 'cup',
  //   ingredient: 'all-purpose flour',
  //   preparation: 'sifted'
  // }
  2. Normalize ingredient name:
  const normalized = parsed.ingredient.toLowerCase().trim();
  // "all-purpose flour"
  3. Check cache first:
  SELECT category, aisle FROM ingredient_categories
  WHERE ingredient = 'all-purpose flour';
  4. If not in cache, call Spoonacular:
  async function categorizeIngredient(ingredientName) {
    // Check cache
    const cached = await db.query(
      'SELECT category, aisle FROM ingredient_categories WHERE ingredient = $1',
      [ingredientName]
    );

    if (cached.rows.length > 0) {
      return cached.rows[0];
    }

    // Call Spoonacular
    try {
      const searchResp = await axios.get(
        'https://api.spoonacular.com/food/ingredients/search',
        {
          params: {
            query: ingredientName,
            apiKey: process.env.SPOONACULAR_API_KEY,
            number: 1
          }
        }
      );

      if (searchResp.data.results.length === 0) {
        // Fallback if not found
        return { category: 'Other', aisle: null };
      }

      const ingredientId = searchResp.data.results[0].id;

      // Get detailed info
      const infoResp = await axios.get(
        `https://api.spoonacular.com/food/ingredients/${ingredientId}/information`,
        {
          params: { apiKey: process.env.SPOONACULAR_API_KEY }
        }
      );

      const category = infoResp.data.categoryPath
        ? infoResp.data.categoryPath[0]
        : infoResp.data.aisle || 'Other';
      const aisle = infoResp.data.aisle;

      // Cache the result
      await db.query(
        `INSERT INTO ingredient_categories
         (ingredient, category, aisle, spoonacular_id, image_url, source)
         VALUES ($1, $2, $3, $4, $5, 'spoonacular')
         ON CONFLICT (ingredient) DO UPDATE
         SET category = $2, aisle = $3, spoonacular_id = $4, image_url = $5`,
        [
          ingredientName,
          category,
          aisle,
          ingredientId,
          infoResp.data.image
        ]
      );

      return { category, aisle };

    } catch (error) {
      console.error('Spoonacular API error:', error);

      // Fallback to basic categorization
      const fallback = getFallbackCategory(ingredientName);

      // Cache fallback to avoid repeated API failures
      await db.query(
        `INSERT INTO ingredient_categories
         (ingredient, category, aisle, source)
         VALUES ($1, $2, NULL, 'fallback')
         ON CONFLICT (ingredient) DO NOTHING`,
        [ingredientName, fallback]
      );

      return { category: fallback, aisle: null };
    }
  }
  5. Build ParsedIngredient object:
  const parsedIngredient = {
    amount: parsed.quantity?.toString() || null,
    unit: parsed.unit || null,
    ingredient: normalized,
    originalText: originalString,
    notes: parsed.preparation || null,
    category: categoryInfo.category,
    aisle: categoryInfo.aisle
  };

  6. Fallback Categorization

  Basic mapping when Spoonacular fails or quota exceeded:

  function getFallbackCategory(ingredient) {
    const categoryMap = {
      // Produce
      'tomato': 'Produce', 'onion': 'Produce', 'garlic': 'Produce',
      'lettuce': 'Produce', 'carrot': 'Produce', 'potato': 'Produce',
      'apple': 'Produce', 'banana': 'Produce', 'lemon': 'Produce',

      // Meat & Seafood
      'chicken': 'Meat', 'beef': 'Meat', 'pork': 'Meat',
      'salmon': 'Seafood', 'shrimp': 'Seafood', 'fish': 'Seafood',

      // Dairy
      'milk': 'Dairy', 'butter': 'Dairy', 'cheese': 'Dairy',
      'yogurt': 'Dairy', 'cream': 'Dairy', 'egg': 'Dairy',

      // Baking
      'flour': 'Baking', 'sugar': 'Baking', 'baking powder': 'Baking',
      'baking soda': 'Baking', 'yeast': 'Baking', 'vanilla': 'Baking',

      // Pantry
      'rice': 'Pantry', 'pasta': 'Pantry', 'oil': 'Pantry',
      'salt': 'Pantry', 'pepper': 'Pantry', 'sauce': 'Pantry',

      // Spices
      'cinnamon': 'Spices', 'cumin': 'Spices', 'paprika': 'Spices',
      'oregano': 'Spices', 'basil': 'Spices', 'thyme': 'Spices',
    };

    const lowerIngredient = ingredient.toLowerCase();

    // Check for matches
    for (const [key, category] of Object.entries(categoryMap)) {
      if (lowerIngredient.includes(key)) {
        return category;
      }
    }

    return 'Other';
  }

  9. Error Handling

  Spoonacular API Errors

  Rate Limit Exceeded (429):
  if (error.response?.status === 429) {
    console.warn('Spoonacular quota exceeded, using fallback');
    return getFallbackCategory(ingredient);
  }

  Invalid API Key (401/403):
  if (error.response?.status === 401 || error.response?.status === 403) {
    console.error('Invalid Spoonacular API key');
    return getFallbackCategory(ingredient);
  }

  Network Error:
  try {
    // Spoonacular call
  } catch (error) {
    console.error('Spoonacular API unavailable:', error.message);
    return getFallbackCategory(ingredient);
  }

  10. Performance & Caching Strategy

  Cache Hit Rate Goals

  - Week 1: ~20% (most ingredients are new)
  - Month 1: ~60% (common ingredients cached)
  - Month 3: ~85% (vast majority cached)
  - Month 6+: ~95% (API calls rare)

  Monitoring

  Track in application logs:
  console.log('Ingredient categorization:', {
    ingredient: name,
    source: 'cache' | 'spoonacular' | 'fallback',
    apiCallsToday: count
  });

  Quota Management

  - Free tier: 150 calls/day
  - Cache lookups: unlimited (database)
  - Expected: 10-30 API calls/day after first month
